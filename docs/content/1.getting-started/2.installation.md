title: Installation

::steps
### Add the dependency

```bash [Terminal]
pnpm add @outloud/css
```

### Use the preset

```ts [uno.config.ts]
import { defineConfig, transformerDirectives, type Preset } from 'unocss'
import { presetOutloud, type Theme } from '@outloud/css'

export default defineConfig<Theme>({
  presets: [presetOutloud({})],
  transformers: [transformerDirectives()],
})
```

### Import the reset styles

```css [styles.css]
@import '@outloud/css/reset.css';
```
